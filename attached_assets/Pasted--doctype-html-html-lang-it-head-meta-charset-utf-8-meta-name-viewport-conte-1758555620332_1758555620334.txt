  <div class="carousel" aria-roledescription="carousel">
    <div class="viewport" id="viewport">
      <button class="arrow prev" id="prevBtn" aria-label="Slide precedente">‹</button>
      <button class="arrow next" id="nextBtn" aria-label="Slide successiva">›</button>

      <div class="track" id="track">
        <div class="slide"><img src="carosello-1.svg" alt="Slide 1"></div>
        <div class="slide"><img src="carosello-2.svg" alt="Slide 2"></div>
        <div class="slide"><img src="carosello-3.svg" alt="Slide 3"></div>
      </div>
    </div>

    <div class="dots" id="dots"></div>
  </div>

  <script>
    const track = document.getElementById('track');
    const slides = Array.from(track.children);
    const prevBtn = document.getElementById('prevBtn');
    const nextBtn = document.getElementById('nextBtn');
    const dotsWrap = document.getElementById('dots');

    let index = 0;
    const last = slides.length - 1;

    slides.forEach((_, i) => {
      const b = document.createElement('button');
      b.className = 'dot';
      b.type = 'button';
      b.addEventListener('click', () => goTo(i));
      dotsWrap.appendChild(b);
    });

    const dots = Array.from(dotsWrap.children);

    function update() {
      track.style.transform = `translateX(${index * -100}%)`;
      prevBtn.disabled = index === 0;
      nextBtn.disabled = index === last;
      dots.forEach((d, i) => d.setAttribute('aria-current', i === index));
    }

    function goTo(i) { index = Math.max(0, Math.min(last, i)); update(); }
    function next() { goTo(index + 1); }
    function prev() { goTo(index - 1); }

    nextBtn.addEventListener('click', next);
    prevBtn.addEventListener('click', prev);

    document.addEventListener('keydown', e => {
      if (e.key === 'ArrowRight') next();
      if (e.key === 'ArrowLeft') prev();
    });

    update();
  </script>
